samples <- rstan::extract(m_full)

pred_dat <- tibble(pre_crastinate = c(-0.5, 0.5))

X = as.matrix(pred_dat)
X <- cbind(1, X)

# Look at prior predictions for mmu
mu <- array(0, dim = c(100000, nrow(X)))
for (ii in 1:100000) {
    # generate a random beta from priors
    beta <- as.vector(c(

        rnorm(1, mean=0, sd=.1)

    mu[ii, ] <- plogis(X %*% beta)
